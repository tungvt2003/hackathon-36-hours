        // Micro-interaction for car movement simulation
        let progress = 0;
        const carIcon = document.querySelector('g[transform^="translate(145"]');
        
        function animateCar() {
            progress += 0.001;
            if (progress > 1) progress = 0;
            // Simple visual jitter to make it look "live"
            const jitterX = Math.sin(Date.now() / 500) * 0.5;
            const jitterY = Math.cos(Date.now() / 500) * 0.5;
            if (carIcon) {
                carIcon.setAttribute('transform', `translate(${145 + jitterX}, ${245 + jitterY})`);
            }
            requestAnimationFrame(animateCar);
        }
        animateCar();

        // Voice button click feedback
        const micBtn = document.querySelector('.voice-pulse');
        micBtn.addEventListener('click', () => {
            const originalColor = micBtn.style.backgroundColor;

                micBtn.style.backgroundColor = '';
