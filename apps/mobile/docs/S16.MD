        function updateStars(rating) {
            const starButtons = document.querySelectorAll('button span.material-symbols-outlined');
            const ratingLabel = document.getElementById('rating-label');
            
            const labels = {
                1: "Chưa tốt ☹️",
                2: "Tạm được 😐",
                3: "Bình thường 🙂",
                4: "Rất tốt! 😊",
                5: "Tuyệt vời! ⭐"
            };

            starButtons.forEach((star, index) => {
                if (index < rating) {
                    star.classList.add('star-filled');
                } else {
                    star.classList.remove('star-filled');
                }
            });

            ratingLabel.innerText = labels[rating] || "";
            ratingLabel.style.color = rating >= 4 ? '#006e2e' : '#575e70';
        }

        function toggleChip(element) {
            element.classList.toggle('chip-active');
            if (element.classList.contains('chip-active')) {
                element.classList.remove('bg-surface', 'border-outline-variant');
            } else {
                element.classList.add('bg-surface', 'border-outline-variant');
            }
        }

        // Simulating the "AI Listening" effect with randomized bars
        const bars = document.querySelectorAll('.visualizer-bar');
        setInterval(() => {
            bars.forEach(bar => {
                const randomHeight = Math.floor(Math.random() * 20) + 4;
                bar.style.height = `${randomHeight}px`;
            });
        }, 150);
    </script>
