<script>
        let isListening = false;
        const voiceOverlay = document.getElementById('voice-overlay');

        function toggleVoice() {
            isListening = !isListening;
            if (isListening) {
                voiceOverlay.classList.remove('opacity-0', 'pointer-events-none');
                voiceOverlay.classList.add('opacity-100');
            } else {
                voiceOverlay.classList.add('opacity-0', 'pointer-events-none');
                voiceOverlay.classList.remove('opacity-100');
            }
        }

        // Add event listener to the floating mic
        document.querySelector('.voice-pulse').parentElement.addEventListener('click', toggleVoice);
